ğŸ”¹ Rest Parameters (Sum of numbers)
function sumAll(...numbers) {
  let sum = 0;
  for (let n of numbers) {
    sum += n;
  }
  return sum;
}

console.log(sumAll(1, 2, 3, 4, 5)); // 15
