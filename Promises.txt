function getUserData(userId) {
  return new Promise((resolve, reject) => {
    if (userId === 1) {
      resolve({ name: "Akshay", age: 25, city: "Pune" });
    } else if (userId === 0) {
      reject("Invalid User");
    }
  });
}

getUserData(1)
  .then(data => {
    console.log("Success:", data);
  })
  .catch(error => {
    console.log("Error:", error);
  });
